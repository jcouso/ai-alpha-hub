## 🧑‍💻 Coding Tips
### 1) Add model deprecation guardrails to your repo today
```bash
#!/usr/bin/env bash
# scan-models.sh
set -euo pipefail

grep -RInE "gemini-3-pro|gpt-5\.1(-codex|-codex-mini|-codex-max)?" . \
  --exclude-dir=node_modules --exclude-dir=.git || true

echo "If matches appear, migrate to: gemini-3.1-pro / gpt-5.3-codex"
```

### 2) Simple routed model fallback (TypeScript)
```ts
type Candidate = { name: string; enabled: boolean };

const candidates: Candidate[] = [
  { name: "gpt-5.3-codex", enabled: true },
  { name: "grok-code-fast-1", enabled: true },
  { name: "gemini-3.1-pro", enabled: true },
];

export function pickModel(preferred?: string) {
  if (preferred) {
    const hit = candidates.find(c => c.name === preferred && c.enabled);
    if (hit) return hit.name;
  }
  return candidates.find(c => c.enabled)?.name ?? "gpt-5.3-codex";
}
```
